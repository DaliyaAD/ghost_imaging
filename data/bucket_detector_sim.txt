# BUCKET SIMULATION
# I need to compare a chosen phantom image with one of the random arrays to get
# an understanding of how accurate that random array was to the O.G image

import numpy as np
import matplotlib.pyplot as plt
# Centre for phantom gen
CENTRE = 16
# Array size
SIZE = 32
# Number of random patterns
M = 100


def grid_size():
    y, x = np.meshgrid(np.arange(SIZE), np.arange(SIZE), indexing='ij')
    return x, y
    # Flip axes to convention

# PHANTOM


def array_sq(width):
    HALF = width//2
    arr_sq = np.zeros((SIZE, SIZE))
    arr_sq[CENTRE-HALF:CENTRE+HALF, CENTRE-HALF:CENTRE+HALF] = 1
    return arr_sq


def array_circ(diameter):
    # Equation of a circle. We can set values above or below as 1 or 0
    x, y = grid_size()
    circle = np.sqrt((x-CENTRE)**2+(y-CENTRE)**2)
    arr_circ = (circle <= diameter).astype(int)
    return arr_circ


def array_ring(diameter):
    x, y = grid_size()
    circle = np.sqrt((x-CENTRE)**2+(y-CENTRE)**2)
    arr_ring = ((diameter-1 <= circle) & (circle <= diameter+1)).astype(int)
    return arr_ring


def array_elp(maj_ax, min_ax):
    x, y = grid_size()
    ellipse = np.sqrt(((x-CENTRE)/min_ax)**2+((y-CENTRE)/maj_ax)**2)
    arr_elp = (ellipse <= 1).astype(int)
    return arr_elp


def array_sin(AMPLITUDE, FREQUENCY, THICKNESS, offset=CENTRE):
    x, y = grid_size()
    sin_wave = AMPLITUDE*np.sin(x*FREQUENCY)+offset
    arr_sin = (np.abs(y - sin_wave) <= THICKNESS).astype(int)
    return arr_sin


def disp_pattern(pattern, file_name):
    plt.imshow(pattern, cmap='gray')
    plt.axis('off')
    plt.savefig(file_name, dpi=400)
    plt.show()


# Random arrays
def array_rand():
    arr_rand = np.random.randint(0, 2, size=(SIZE, SIZE))
    return arr_rand

# 3D array that stacks M random patterns


def many_array_rand():
    many_rand = np.array([array_rand() for i in range(M)])
    return many_rand

# As these are currently either 1 or 0, I can multiply the elements together
# and sum over the number of 'bright' pixels. This will act as my weighting factor


def rand_phant_prod(random, phantom):
    prod = random*phantom
    return prod


stack = many_array_rand()

pattern = array_ring(10)
# disp_pattern(pattern)
arr_weight = np.zeros(M)

# I now need to assign the random pattern a weight based on the "correct" number of pixels
# The stack of M arrays is 3D so I need to pick out each 'slice' iteratively and
# compare to the phantom

for i in range(M):
    bucket = rand_phant_prod(stack[i], pattern)
    weight = np.sum(bucket)/(SIZE*SIZE)
    arr_weight[i] = weight
    # disp_pattern(bucket)
# I now have an array for the weighting of each random pattern
# I can plot this against pattern number to show how this fluctuates

mean_weight = np.mean(arr_weight)

# Plotting details

def create_plot(x_data, y_data):
    # Main plot
    figure = plt.figure(figsize=(8, 9))
    plt.rcParams['font.family'] = 'Times New Roman'
    axes_main_plot = figure.add_subplot(211)
    axes_main_plot.grid(GRID_LINES)
    axes_main_plot.xaxis.set_major_locator(plt.MultipleLocator(M/5))
    axes_main_plot.scatter(x_data, y_data)
    axes_main_plot.set_title(PLOT_TITLE, fontsize=14)
    axes_main_plot.set_xlabel(X_LABEL, fontsize=18)
    axes_main_plot.set_ylabel(Y_LABEL, fontsize=18)
    axes_main_plot.tick_params(axis='both', labelsize=12)


# Now I want to use the weighting to 'add' together all of the random patterns
# based on how true they are to the original picture
# Then, I can see how this accuracy changes for different values of M
# I have the weightings in an array, I will need to pick out the scalar value
# in order to weigh the random patterns


def pixel_weighting():
    weighted_pattern = np.zeros((SIZE, SIZE))
    for i in range(M):
        weighted_pattern = weighted_pattern + \
            stack[i]*(arr_weight[i]-mean_weight)
    disp_pattern(weighted_pattern, "Ring_100.png")

pixel_weighting()
