import numpy as np
import matplotlib.pyplot as plt

CENTRE = 16
SIZE = 32
M = 1000


def array_rand():
    arr_rand = np.random.randint(0, 2, size=(SIZE, SIZE))
    return arr_rand


def many_array_rand():
    many_rand = np.array([array_rand() for i in range(M)])
    return many_rand


def disp_pattern(pattern):
    plt.imshow(pattern, cmap='gray')
    plt.axis('off')
    plt.show()
    print(pattern)


def save_array(stack):
    np.save("Many_array_stack", stack)


stack = many_array_rand()
disp_pattern(stack[0])
